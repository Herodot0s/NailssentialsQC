**2. Add package validation** before creating appointment items. For each unique `packageId` in the items array:
```typescript
const packageIds = [...new Set(items.filter(i => i.packageId).map(i => i.packageId!))];
for (const pkgId of packageIds) {
  const pkg = await prisma.servicePackage.findUnique({
    where: { id: pkgId },
    include: { appointment_items: { select: { id: true } } }
  });

  if (!pkg) {
    return res.status(400).json({ success: false, message: `Package ${pkgId} not found` });
  }
  if (!pkg.is_active) {
    return res.status(400).json({ success: false, message: `Package "${pkg.name}" is no longer available` });
  }

  // Validity date check
  const now = new Date();
  if (pkg.valid_from && new Date(pkg.valid_from) > now) {
    return res.status(400).json({ success: false, message: `Package "${pkg.name}" is not yet available` });
  }
  if (pkg.valid_until && new Date(pkg.valid_until) < now) {
    return res.status(400).json({ success: false, message: `Package "${pkg.name}" has expired` });
  }

  // Max redemptions check
  if (pkg.max_redemptions !== null) {
    const currentBookings = pkg.appointment_items.length;
    const newBookingsForThisPkg = items.filter(i => i.packageId === pkgId).length;
    if (currentBookings + newBookingsForThisPkg > pkg.max_redemptions) {
      return res.status(400).json({ success: false, message: `Package "${pkg.name}" has reached its booking limit` });
    }
  }
}
```

**1. Calculate `Transaction.amount` correctly for packages (D-07):**

Replace the existing `totalAmount` calculation (which sums `price_at_booking` of all items) with logic that accounts for packages:

```typescript
// Group items by package_id
const packageGroups = new Map<number, typeof appointment.items>();
const individualItems: typeof appointment.items = [];

for (const item of appointment.items) {
  if (item.package_id) {
    if (!packageGroups.has(item.package_id)) {
      packageGroups.set(item.package_id, []);
    }
    packageGroups.get(item.package_id)!.push(item);
  } else {
    individualItems.push(item);
  }
}

// Calculate actual paid amount
let totalAmount = new Decimal(0);

// Individual items: use price_at_booking
for (const item of individualItems) {
  totalAmount = totalAmount.plus(item.price_at_booking);
}

// Package items: use package price (not sum of individual prices)
for (const [packageId, items] of packageGroups) {
  const pkg = await prisma.servicePackage.findUnique({ where: { id: packageId } });
  if (pkg) {
    totalAmount = totalAmount.plus(pkg.price);  // Package price = what customer paid
  } else {
    // Fallback: sum individual prices if package not found
    for (const item of items) {
      totalAmount = totalAmount.plus(item.price_at_booking);
    }
  }
}
```
