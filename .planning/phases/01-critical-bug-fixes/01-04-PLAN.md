1. Add Zod import:
```typescript
import { z } from 'zod';
```

2. Add idParamSchema and validateIdParam middleware:
```typescript
const idParamSchema = z.object({
  id: z.string().regex(/^\d+$/, 'ID must be a number').transform(Number)
});

const validateIdParam = (req: any, res: any, next: any) => {
  const result = idParamSchema.safeParse(req.params);
  if (!result.success) {
    return res.status(400).json({
      success: false,
      error: { code: 'INVALID_PARAMETER', message: 'Invalid ID parameter' }
    });
  }
  req.validatedParams = result.data;
  next();
};
```

3. Add validateIdParam to all routes with :id param:
- payrollRoutes.ts: `/periods/:id`, `/periods/:id/lock`
- attendanceRoutes.ts: `/:id`

Example for payrollRoutes.ts:

router.patch('/periods/:id/lock', authenticateToken, authorizeRoles('manager'), validateIdParam, lockPayroll);
